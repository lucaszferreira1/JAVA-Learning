Para fazer funcionar o programa é preciso adicionar a versão mais nova do postgresql.jar ao build path do programa.



Abaixo estará as querys de criação das tabelas do BD;

CREATE TABLE BARCO (
	ID SERIAL PRIMARY KEY,
	NOME VARCHAR(255),
	TIPO VARCHAR(255)
)

CREATE TABLE FUNCIONARIO(
	ID SERIAL PRIMARY KEY,
	NOME VARCHAR(255) NOT NULL,
	TRABALHO VARCHAR(255),
	EMAIL VARCHAR(255),
	FONE VARCHAR(32),
	CPF INT NOT NULL
)

CREATE TABLE CAPITAO(
	ID_BARCO INT NOT NULL,
	ID_CAPITAO INT NOT NULL,
	PRIMARY KEY(ID_BARCO, ID_CAPITAO),
	FOREIGN KEY(ID_BARCO)
		REFERENCES BARCO (ID),
	FOREIGN KEY(ID_CAPITAO)
		REFERENCES FUNCIONARIO (ID)
)
